# AI Agent Frontend Components - Usage Guide\n\n## Overview\n\nThis comprehensive frontend enhancement provides a complete AI-powered content generation system with multiple components for different use cases.\n\n## Components Overview\n\n### 1. AIIntegration (Recommended Entry Point)\n\n**Purpose**: Drop-in component that can be added to any existing form or interface\n**Best for**: Quick integration into existing course creation flows\n\n```tsx\nimport { AIIntegration } from '@/components/instructor/ai-agent';\n\n// In any course creation component\n<AIIntegration\n  courseId={courseId}\n  moduleId={moduleId}\n  triggerText=\"Generate with AI\"\n  onContentGenerated={(type, data) => {\n    // Handle the generated content\n    console.log('Generated:', type, data);\n  }}\n/>\n```\n\n### 2. AIDashboard (Full Featured)\n\n**Purpose**: Complete dashboard with health monitoring, recent generations, and settings\n**Best for**: Dedicated AI generation pages\n\n```tsx\nimport { AIDashboard } from '@/components/instructor/ai-agent';\n\n<AIDashboard\n  courseId={courseId}\n  moduleId={moduleId}\n  onContentGenerated={(type, data) => {\n    // Handle generated content\n  }}\n/>\n```\n\n### 3. EnhancedAIContentGenerator (Advanced)\n\n**Purpose**: Advanced generator with progress tracking and quality assessment\n**Best for**: When you need granular control over the generation process\n\n```tsx\nimport { EnhancedAIContentGenerator } from '@/components/instructor/ai-agent';\n\n<EnhancedAIContentGenerator\n  type=\"lesson\" // 'course' | 'module' | 'lesson' | 'quiz'\n  courseId={courseId}\n  moduleId={moduleId}\n  onGenerate={(data) => {\n    // Handle generated data\n  }}\n/>\n```\n\n### 4. AIAdminPanel (Admin Only)\n\n**Purpose**: Admin configuration and monitoring\n**Best for**: System administration pages\n\n```tsx\nimport { AIAdminPanel } from '@/components/instructor/ai-agent';\n\n// In admin dashboard\n<AIAdminPanel\n  onConfigUpdate={() => {\n    // Handle configuration updates\n  }}\n/>\n```\n\n## Integration Examples\n\n### Example 1: Course Creation Page\n\n```tsx\n\"use client\";\n\nimport React, { useState } from 'react';\nimport { AIIntegration } from '@/components/instructor/ai-agent';\nimport { Button } from '@/components/ui/button';\nimport { Input } from '@/components/ui/input';\n\nexport default function CreateCoursePage() {\n  const [courseData, setCourseData] = useState({\n    title: '',\n    description: '',\n    modules: []\n  });\n\n  const handleAIGeneration = (type: string, data: any) => {\n    if (type === 'course') {\n      setCourseData({\n        title: data.title,\n        description: data.description,\n        modules: data.modules\n      });\n    }\n  };\n\n  return (\n    <div className=\"space-y-6\">\n      <div className=\"flex justify-between items-center\">\n        <h1 className=\"text-2xl font-bold\">Create New Course</h1>\n        \n        {/* AI Integration Button */}\n        <AIIntegration\n          triggerText=\"Generate Course with AI\"\n          onContentGenerated={handleAIGeneration}\n        />\n      </div>\n      \n      <form className=\"space-y-4\">\n        <Input\n          placeholder=\"Course Title\"\n          value={courseData.title}\n          onChange={(e) => setCourseData(prev => ({ ...prev, title: e.target.value }))}\n        />\n        \n        <textarea\n          placeholder=\"Course Description\"\n          value={courseData.description}\n          onChange={(e) => setCourseData(prev => ({ ...prev, description: e.target.value }))}\n          className=\"w-full p-3 border rounded-lg\"\n          rows={4}\n        />\n        \n        {/* Show generated modules if available */}\n        {courseData.modules.length > 0 && (\n          <div>\n            <h3 className=\"font-semibold mb-2\">Generated Modules:</h3>\n            <ul className=\"space-y-2\">\n              {courseData.modules.map((module: any, index: number) => (\n                <li key={index} className=\"p-3 border rounded-lg\">\n                  <h4 className=\"font-medium\">{module.title}</h4>\n                  <p className=\"text-sm text-muted-foreground\">{module.description}</p>\n                </li>\n              ))}\n            </ul>\n          </div>\n        )}\n        \n        <Button type=\"submit\">Create Course</Button>\n      </form>\n    </div>\n  );\n}\n```\n\n### Example 2: Module Creation with AI\n\n```tsx\n\"use client\";\n\nimport React from 'react';\nimport { AIIntegration } from
